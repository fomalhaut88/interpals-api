 96    def search(self, options: dict, limit: int = 1000, 
 97               timeout: float = 0.0) -> Generator[str, None, None]:
 98        """
 99        Search for users according to the search `options`. The total number
100        of users is `limit`. Since users are requested as pages, `timeout`
101        is a timeout between page loading.
102        """
103        response = self._get("/app/search", check_auth=True)
104        csrf_token = find_csrf_token(response.text)
105
106        offset = 0
107        params = self._prepare_search_params(options)
108        params['csrf_token'] = csrf_token
109        while True:
110            params['offset'] = str(offset)
111
112            response = self._get("/app/search", params, check_auth=True)
113            users = self._parse_search_result(response.text)
114
115            if not users:
116                return
117
118            for user in users:
119                yield user
120                offset += 1
121                if offset >= limit:
122                    return
123
124            sleep(timeout)

146    def chat(self, count: int = 9, offset: int = 0) -> Dict[str, Any]:
147        """
148        Get general chat information in count `count` with the offset `offset`.
149        """
150        chat_parser = ChatParser()
151
152        response = self._get("/pm.php", check_auth=True)
153        maxmsgid = chat_parser.parse_maxmsgid(response.text)
154        unread = chat_parser.parse_unread(response.text)
155
156        chats = []
157        while len(chats) < count:
158            params = {
159                'action': 'more_threads',
160                'from': str(offset),
161                'filter': 'all',
162                'max_msg_id': maxmsgid,
163            }
164            response = self._post("/pm.php", params)
165            body = response.json()['body']
166
167            items = chat_parser.parse_chat(body)
168            if not items:
169                break
170
171            for item in items:
172                chats.append(item)
173                if len(chats) >= count:
174                    break
175
176            offset += len(items)
177
178        return {
179            'chats': chats,
180            'unread': unread
181        }

277    @classmethod
278    def _get_full_url(cls, url, params=None):
279        if not url.startswith("/"):
280            url = "/" + url
281        url = "https://{}{}".format(cls.host, url)
282        if params is not None:
283            url += '?' + urlencode(params, True)
284        return url
285
286    def _get(self, url, params=None, check_auth=False):
287        headers = self._get_headers()
288        if params is not None:
289            url = url + '?' + urlencode(params, True)
290        full_url = self._get_full_url(url)
291        response = requests.get(full_url, headers=headers, timeout=self.timeout, 
292                                allow_redirects=False)
293        if check_auth and not self._check_body_for_auth(response.text):
294            raise APIAuthError()
295        return response
296
297    def _post(self, url, params, check_auth=False):
298        headers = self._get_headers()
299        headers['Content-Type'] = 'application/x-www-form-urlencoded'
300        full_url = self._get_full_url(url)
301        response = requests.post(full_url, data=params, headers=headers, 
302                                 timeout=self.timeout, allow_redirects=False)
303        if check_auth and not self._check_body_for_auth(response.text):
304            raise APIAuthError()
305        return response

319    def _prepare_search_params(self, options):
320        params = {
321            'offset': '',
322            'sort': 'last_login',
323            'age1': options.get('age1', '16'),
324            'age2': options.get('age2', '110'),
325            'sex[]': options.get('sex', ['male', 'female']),
326            'continents[]': options.get('continents', ['AF', 'AS', 'EU', 'NA', 
327                                                       'OC', 'SA']),
328            'countries[]': options.get('countries', ['---']),
329            'languages[]': ['---'],
330            'lfor[]': [
331                'lfor_email', 'lfor_snail', 'lfor_langex',
332                'lfor_friend', 'lfor_flirt', 'lfor_relation'
333            ],
334            'keywords': options.get('keywords', ''),
335            'username': '',
336            'csrf_token': 'ZjU1ZWZkM2Q=',
337        }
338
339        if options.get('online'):
340            params['online'] = '1'
341
342        if options.get('cityName'):
343            citycode = options.get('city')
344            cityname = options.get('cityName')
345            if not citycode:
346                citycode = self._get_citycode(cityname)
347            params['city'] = citycode
348            params['cityName'] = cityname
349
350        return params

624    async def _request(self, url, method='get', params=None, check_auth=True):
625        assert method in ('get', 'post')
626
627        full_url = self._get_full_url(url, params if method == 'get' else None)
628        headers = self._get_headers()
629
630        if method == 'post':
631            headers['Content-Type'] = 'application/x-www-form-urlencoded'
632
633        kwargs = {
634            'url': full_url,
635            'headers': headers,
636            'timeout': self.timeout,
637            'allow_redirects': False,
638        }
639
640        if method == 'post':
641            kwargs['data'] = params
642
643        async with aiohttp.ClientSession() as session:
644            request_func = session.post if method == 'post' else session.get
645            async with request_func(**kwargs) as resp:
646                if resp.status in (301, 302):
647                    raise APIRedirectError(
648                        resp.status, resp.headers['Location']
649                    )
650                body = await resp.text()
651
652        if check_auth and not self._check_body_for_auth(body):
653            raise APIAuthError()
654        
655        return body

287    def _get(self, url, params=None, check_auth=False):
288        headers = self._get_headers()

291        full_url = self._get_full_url(url)
292        response = requests.get(full_url, headers=headers, timeout=self.timeout, 
293                                allow_redirects=False)

295            raise APIAuthError()

625    async def _request(self, url, method='get', params=None, check_auth=True):
626        assert method in ('get', 'post')
627
628        full_url = self._get_full_url(url, params if method == 'get' else None)
629        headers = self._get_headers()
630
631        if method == 'post':
632            headers['Content-Type'] = 'application/x-www-form-urlencoded'
633
634        kwargs = {
635            'url': full_url,
636            'headers': headers,
637            'timeout': self.timeout,
638            'allow_redirects': False,
639        }
640
641        if method == 'post':
642            kwargs['data'] = params
643
644        async with aiohttp.ClientSession() as session:
645            request_func = session.post if method == 'post' else session.get
646            async with request_func(**kwargs) as resp:
647                if resp.status in (301, 302):
648                    raise APIRedirectError(
649                        resp.status, resp.headers['Location']
650                    )
651                body = await resp.text()
652
653        if check_auth and not self._check_body_for_auth(body):
654            raise APIAuthError()
